Task 1

За допомогою властивостей select.options, select.value та select.selectedIndex, додати 4 функціональні кнопки для навігації по елементам випадаючого списку.



let select = document.getElementById('select');
let newElementText = document.getElementById('newElementText');
function addElement() {
    if (newElementText.value !== '') {
        const option = document.createElement('option');
        option.innerText = newElementText.value;
        select.appendChild(option);
        newElementText.value = ''
    }
}
function getFirst() {
    select.selectedIndex = 0;
}
function getPrev() {
    if (select.selectedIndex !== 0) {
        select.selectedIndex--;
    }
}
function getNext() {
    if (select.selectedIndex !== select.length - 1) {
        select.selectedIndex++;
    }
}
function getLast() {
    select.selectedIndex = select.length - 1;
}
